```python
def make_focal_loss(alpha=0.75, gamma=2.0):
    def focal_loss(preds, dtrain):
        labels = dtrain.get_label()
        p = 1.0 / (1.0 + np.exp(-preds))  # sigmoid
        
        grad = np.zeros_like(p)
        hess = np.ones_like(p)  # Hessian ê·¼ì‚¬ (=1)
        
        # y=1 (rare) ìƒ˜í”Œ
        idx_pos = (labels == 1)
        grad[idx_pos] = alpha * (1.0 - p[idx_pos]) ** gamma * (
            gamma * p[idx_pos] * np.log(p[idx_pos] + eps) + p[idx_pos] - 1.0
        )
        
        # y=0 (others) ìƒ˜í”Œ
        idx_neg = (labels == 0)
        grad[idx_neg] = (1-alpha) * (
            p[idx_neg] ** (gamma + 1.0)
            - gamma * p[idx_neg] ** gamma * (1 - p[idx_neg]) * np.log(1 - p[idx_neg] + eps)
        )
        
        return grad, hess
    return focal_loss
```
